// Optimize API calls by implementing caching and batching requests
import axios from 'axios';
import { cache } from 'memory-cache';
const lineApiUrl = 'https://api.line.me/v2';
const channelId = 'YOUR_CHANNEL_ID';
const channelSecret = 'YOUR_CHANNEL_SECRET';
const api = axios.create({
  baseURL: lineApiUrl,
  headers: {
    'Content-Type': 'application/json',
    'Authorization': `Bearer ${channelSecret}`,
  },
});
const cacheKey = 'clock-in-out-data';
const cacheTtl = 300; // 5 minutes
const getData = async () => {
  const cachedData = cache.get(cacheKey);
  if (cachedData) {
    return cachedData;
  }
  const response = await api.get(`/v2/bot/message/reply`, {
    params: {
      replyToken: 'YOUR_REPLY_TOKEN',
    },
  });
  const data = response.data;
  cache.put(cacheKey, data, cacheTtl);
  return data;
};

    }

  return (

          {clockedIn ? 'Clock Out' : 'Clock In'}

    </div>
  );
