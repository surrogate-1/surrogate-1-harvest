```python
import json
import os
import requests

def list_repo_tree(repo, path, recursive=False):
    api_key = os.environ['HF_API_KEY']
    url = f'https://huggingface.co/api/v1/repos/{repo}/tree/{path}?recursive={recursive}'
    response = requests.get(url, headers={'Authorization': f'Bearer {api_key}'})
    file_paths = response.json()['tree']
    return [file['path'] for file in file_paths]

def pre_list_file_paths(repo, date, folder):
    file_paths = list_repo_tree(repo, f'{date}/{folder}')
    with open('file_paths.json', 'w') as f:
        json.dump(file_paths, f)
    return file_paths

# Usage
repo = 'your-repo-name'
date = 'your-date'
folder = 'your-folder'
file_paths = pre_list_file_paths(repo, date, folder)
```
