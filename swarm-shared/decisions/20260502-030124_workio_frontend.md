```typescript
// Optimize clock in/out feature
import { useState, useEffect } from 'react';

const ClockInOut = () => {
  const [location, setLocation] = useState(null);

  useEffect(() => {
    // Get current location
    navigator.geolocation.getCurrentPosition((position) => {
      setLocation(position.coords);
    });
  }, []);

  const handleClockIn = () => {
    // Verify location and clock in
    if (location) {
      // Call API to clock in
      fetch('/api/clock-in', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          location: {
            latitude: location.latitude,
            longitude: location.longitude,
          },
        }),
      })
      .then((response) => response.json())
      .then((data) => console.log(data))
      .catch((error) => console.error(error));
    }
  };

  return (
    <div>
      <button onClick={handleClockIn}>Clock In</button>
    </div>
  );
};

export default ClockInOut;
```

```javascript
// Update LINE OA webhook
const express = require('express');
const app = express();

app.post('/webhook/line', (req, res) => {
  // Handle clock in/out events from LINE OA
  const eventType = req.body.events[0].type;
  if (eventType === 'clock_in' || eventType === 'clock_out') {
    // Call API to handle clock in/out event
    fetch('/api/handle-clock-event', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(req.body),
    })
    .then((response) => response.json())
    .then((data) => res.json(data))
    .catch((error) => res.status(500).json({ error: 'Internal Server Error' }));
  }
});

app.listen(3000, () => {
  console.log('LINE OA webhook listening on port 3000');
});
```
