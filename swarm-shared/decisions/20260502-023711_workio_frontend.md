// Import necessary dependencies
import React, { useState, useEffect } from 'react';
import { useKnowledgeRag } from '../hooks/useKnowledgeRag';

// Create a new component to display the most-connected hub
const MostConnectedHub = () => {
  const [hub, setHub] = useState(null);
  const { data, error } = useKnowledgeRag();

  useEffect(() => {
    if (data) {
      const mostConnectedHub = data.hubs.find((hub) => hub.name === 'MOC');
      setHub(mostConnectedHub);
    }
  }, [data]);

  if (error) {
    return <div>Error: {error.message}</div>;
  }

  if (!hub) {
    return <div>Loading...</div>;
  }

  return (
    <div>
      <h2>Most Connected Hub: {hub.name}</h2>
      <p>Related Documents: {hub.relatedDocuments.join(', ')}</p>
    </div>
  );
};

// Use the new component in the existing frontend code
const App = () => {
  return (
    <div>
      <MostConnectedHub />
      {/* Existing frontend code */}
    </div>
  );
};
