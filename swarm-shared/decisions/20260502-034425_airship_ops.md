```python
import json
import requests

# Pre-list file paths using a single API call
def pre_list_file_paths(repo_id, path):
    url = f"https://huggingface.co/api/v1/repo/{repo_id}/tree/{path}"
    response = requests.get(url)
    file_paths = response.json()["files"]
    with open("file_paths.json", "w") as f:
        json.dump(file_paths, f)

# Embed the file list in the train.py script
def load_data():
    with open("file_paths.json", "r") as f:
        file_paths = json.load(f)
    # Use CDN-only fetches with zero API calls during data load
    data = []
    for file_path in file_paths:
        url = f"https://huggingface.co/{repo_id}/resolve/main/{file_path}"
        response = requests.get(url)
        data.append(response.content)
    return data
```
