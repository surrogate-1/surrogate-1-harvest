hf_api_list_repo_tree() {
  repo=$1
  path=$2
  curl -X GET \
  https://huggingface.co/api/v1/repo/${repo}/tree/${path} \
  -H 'Authorization: Bearer YOUR_API_TOKEN' \
  -H 'Content-Type: application/json'
}

# Apply HF CDN Bypass pattern
wget https://huggingface.co/datasets/{repo}/resolve/main/{path}
file_paths=$(curl -X GET \
https://huggingface.co/api/v1/datasets/{repo}/tree \
-H 'Authorization: Bearer {token}' \
-H 'Content-Type: application/json')
echo "$file_paths" > file_paths.json
file_paths=$(jq -r '.[] | .path' file_paths.json)
for file_path in $file_paths; do
  wget https://huggingface.co/datasets/{repo}/resolve/main/$file_path
done
