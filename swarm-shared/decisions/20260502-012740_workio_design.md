#### Code Snippets
```typescript
// ClockIn.tsx
import React from 'react';

const ClockIn: React.FC = () => {
  const [clockedIn, setClockIn] = React.useState(false);

  const handleClockIn = () => {
    // Clock in logic
    setClockIn(true);
  };

  const handleClockOut = () => {
    // Clock out logic
    setClockIn(false);
  };

  return (
    <div>
      {clockedIn ? (
        <button onClick={handleClockOut}>Clock Out</button>
      ) : (
        <button onClick={handleClockIn}>Clock In</button>
      )}
    </div>
  );
};

export default ClockIn;
```
