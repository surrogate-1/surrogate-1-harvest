import requests

def download_from_cdn(repo, path, file_name):
    url = f"https://huggingface.co/datasets/{repo}/resolve/main/{path}/{file_name}"
    response = requests.get(url)
    if response.status_code == 200:
        return response.content
    else:
        raise Exception(f"Failed to download file from CDN: {response.status_code}")

def load_data(repo, path):
    # Use CDN downloads for file fetching
    file_name = path.split("/")[-1]
    return download_from_cdn(repo, path, file_name)
