```python
import json

class ServiceRegistry:
    def __init__(self):
        self.services = {}

    def register_service(self, service_name, endpoint, protocol):
        self.services[service_name] = {
            'endpoint': endpoint,
            'protocol': protocol
        }

    def get_service(self, service_name):
        return self.services.get(service_name)

# discovery_mechanism.py
import requests

class DiscoveryMechanism:
    def __init__(self, service_registry):
        self.service_registry = service_registry

    def discover_services(self):
        services = []
        for service_name, service_info in self.service_registry.services.items():
            services.append({
                'name': service_name,
                'endpoint': service_info['endpoint'],
                'protocol': service_info['protocol']
            })
        return services

# example usage
service_registry = ServiceRegistry()
service_registry.register_service('arkship', 'http://localhost:8000', 'http')
service_registry.register_service('surrogate', 'http://localhost:8001', 'http')

discovery_mechanism = DiscoveryMechanism(service_registry)
services = discovery_mechanism.discover_services()
print(services)
```
This code snippet demonstrates a basic implementation of a unified discovery mechanism using a service registry. The `ServiceRegistry` class stores information about each microservice, and the `DiscoveryMechanism` class uses this registry to discover available services.
