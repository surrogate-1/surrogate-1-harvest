### Code Snippets

```python
# Import required libraries
from redis import Redis
from sqlalchemy import create_engine

# Establish Redis connection for caching
redis_client = Redis(host='localhost', port=6379, db=0)

# Define a function to cache service registry data
def cache_service_registry_data(data):
    redis_client.set('service_registry_data', data)

# Define a function to retrieve cached service registry data
def get_cached_service_registry_data():
    return redis_client.get('service_registry_data')

# Optimize database queries using SQLAlchemy
engine = create_engine('postgresql://user:password@host:port/dbname')
query = "SELECT * FROM service_registry"
result = engine.execute(query)

# Implement lazy loading for service registry data
def load_service_registry_data():
    # Load data only when required
    if not get_cached_service_registry_data():
        data = engine.execute(query)
        cache_service_registry_data(data)
    return get_cached_service_registry_data()
```
